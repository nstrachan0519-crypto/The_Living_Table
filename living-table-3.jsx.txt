function currentSeasonId(hemisphere) {
  const m = new Date().getMonth(); // 0-11
  let id;
  if (m >= 2 && m <= 4) id = "spring";
  else if (m >= 5 && m <= 7) id = "summer";
  else if (m >= 8 && m <= 10) id = "fall";
  else id = "winter";
  if (hemisphere === "south") {
    const flip = { spring: "fall", summer: "winter", fall: "spring", winter: "summer" };
    id = flip[id];
  }
  return id;
}

// The engine. Given a list of selected food ids, decide whether the protein is complete,
// and if not, name exactly what would finish it.
function analyzeProtein(selectedIds) {
  const picked = PROTEIN_FOODS.filter((f) => selectedIds.includes(f.id));
  const totalProtein = picked.reduce((s, f) => s + f.protein, 0);
  const groups = new Set(picked.map((f) => f.group));

  const hasComplete = groups.has("complete") || groups.has("animal-complete");
  const hasLegume = groups.has("legume");
  const hasGrainOrSeed = groups.has("grain") || groups.has("nutseed");

  // Which of the two limiting amino acids are covered?
  const lysineOk = hasComplete || hasLegume;
  const methionineOk = hasComplete || hasGrainOrSeed;
  const complete = lysineOk && methionineOk;

  let need = null;
  let suggestions = [];
  if (!picked.length) {
    need = "empty";
  } else if (complete) {
    need = "none";
  } else if (!methionineOk) {
    // they have legumes only — they need a grain, nut, or seed
    need = "methionine";
    suggestions = PROTEIN_FOODS.filter((f) => f.group === "grain" || f.group === "nutseed")
      .sort((a, b) => a.cost - b.cost).slice(0, 6);
  } else {
    // they have grains/nuts/seeds only — they need a legume
    need = "lysine";
    suggestions = PROTEIN_FOODS.filter((f) => f.group === "legume")
      .sort((a, b) => a.cost - b.cost).slice(0, 6);
  }

  // A single complete food always finishes the job too, whatever is missing.
  const shortcuts = PROTEIN_FOODS.filter((f) => f.group === "complete")
    .sort((a, b) => a.cost - b.cost).slice(0, 4);

  // Rough cost of the plate, and cost per 10 g of protein.
  const cost = picked.reduce((s, f) => s + f.cost, 0);
  const costPer10 = totalProtein > 0 ? (cost / totalProtein) * 10 : 0;

  return { picked, totalProtein, complete, need, suggestions, shortcuts, cost, costPer10, lysineOk, methionineOk };
}

function computeResults(profile) {
  const w = parseFloat(profile.weight);
  const h = parseFloat(profile.height);
  const a = parseFloat(profile.age);
  if (!w || !h || !a) return null;

  const male = 10 * w + 6.25 * h - 5 * a + 5;
  const female = 10 * w + 6.25 * h - 5 * a - 161;
  let bmr;
  if (profile.sex === "male") bmr = male;
  else if (profile.sex === "female") bmr = female;
  else bmr = (male + female) / 2;

  const activity = ACTIVITY.find((x) => x.id === profile.activity) || ACTIVITY[0];
  const goal = GOALS.find((x) => x.id === profile.goal) || GOALS[0];

  const tdee = bmr * activity.mult;
  const calories = Math.max(1200, Math.round(tdee + goal.cals));
  const proteinG = Math.round(w * goal.protein);
  const proteinCals = proteinG * 4;
  const fatCals = calories * 0.28;
  const fatG = Math.round(fatCals / 9);
  const carbCals = calories - proteinCals - fatCals;
  const carbG = Math.round(Math.max(carbCals, 0) / 4);

  return { bmr: Math.round(bmr), tdee: Math.round(tdee), calories, proteinG, fatG, carbG };
}

  function importTraditions(e) {
    const file = e.target.files && e.target.files[0];
    if (!file) return;
    const reader = new FileReader();
    reader.onload = async () => {
      try {
        const parsed = JSON.parse(reader.result);
        const incoming = Array.isArray(parsed) ? parsed : parsed.traditions || [];
        const valid = incoming.filter((t) => t && t.title && t.body);
        if (!valid.length) {
          setTradMessage("That file didn't contain any readable traditions.");
          return;
        }
        const existing = new Set(traditions.map((t) => t.id));
        const fresh = valid.filter((t) => !existing.has(t.id))
          .map((t) => ({
            id: t.id || Date.now() + "-" + Math.random().toString(36).slice(2, 8),
            title: String(t.title).slice(0, 120),
            people: String(t.people || "").slice(0, 120),
            category: TRADITION_CATEGORIES.includes(t.category) ? t.category : "Other",
            body: String(t.body).slice(0, 4000),
            recorded: t.recorded || todayKey,
          }));
        const next = [...fresh, ...traditions].slice(0, 300);
        setTraditions(next);
        setTradMessage(fresh.length
          ? `Welcomed ${fresh.length} tradition${fresh.length === 1 ? "" : "s"} to the table.`
          : "Everything in that file was already here.");
        try { await window.storage.set("living-table-traditions", JSON.stringify(next)); }
        catch (err) { console.error("Could not save imported traditions", err); }
      } catch (err) {
        console.error("Import failed", err);
        setTradMessage("That file couldn't be read — it should be a traditions file exported from this app.");
      }
    };
    reader.readAsText(file);
    e.target.value = "";
  }

function RecipeCard({ recipe, onSave, onAddToShopping, proteinInfoOpen, setProteinInfoOpen }) {
  const n = recipe.nutrition || {};
  return (
    <div className="rounded-2xl p-5 mt-4" style={{ background: COLORS.surface, border: `1px solid ${COLORS.line}` }}>
      <h3 className="display" style={{ fontSize: "1.1rem", fontWeight: 600 }}>{recipe.title}</h3>
      <p style={{ fontSize: "0.8rem", color: COLORS.inkSoft, marginBottom: "0.75rem" }}>
        {recipe.culture} · Serves {recipe.servings} · {(recipe.dietary_tags || []).join(", ")}
      </p>

      <h4 style={{ fontSize: "0.85rem", fontWeight: 600, marginTop: "0.75rem" }}>Ingredients</h4>
      <ul style={{ fontSize: "0.85rem", marginTop: "0.35rem" }} className="space-y-1">
        {(recipe.ingredients || []).map((ing, i) => (
          <li key={i}>{ing.amount} {ing.name}</li>
        ))}
      </ul>

      <h4 style={{ fontSize: "0.85rem", fontWeight: 600, marginTop: "1rem" }}>Steps</h4>
      <ol style={{ fontSize: "0.85rem", marginTop: "0.35rem" }} className="space-y-1.5 list-decimal list-inside">
        {(recipe.steps || []).map((s, i) => <li key={i}>{s}</li>)}
      </ol>

      <h4 style={{ fontSize: "0.85rem", fontWeight: 600, marginTop: "1rem" }}>Nutrition (per serving)</h4>
      <div className="grid grid-cols-2 md:grid-cols-4 gap-2 mt-2">
        <Stat label="Calories" value={n.calories} color={COLORS.ink} />
        <Stat label="Protein (g)" value={n.protein_g} color={COLORS.clay} />
        <Stat label="Carbs (g)" value={n.carbs_g} color={COLORS.gold} />
        <Stat label="Fat (g)" value={n.fat_g} color={COLORS.calm} />
      </div>

      <div className="mt-3 rounded-xl p-3" style={{ background: `${COLORS.calm}18` }}>
        <button onClick={() => setProteinInfoOpen(!proteinInfoOpen)}
          className="flex items-center gap-2 text-sm font-medium">
          <span>{n.protein_complete ? "✓ Complete protein" : "△ Incomplete protein"}</span>
          <Info size={14} color={COLORS.calm} />
        </button>
        {proteinInfoOpen && (
          <div style={{ fontSize: "0.8rem", color: COLORS.inkSoft, marginTop: "0.4rem" }}>
            <p style={{ marginBottom: "0.4rem" }}>
              <strong style={{ color: COLORS.ink }}>What does this mean? </strong>
              Protein is built from little pieces called amino acids. Your body needs 9 of
              them from food — it can't make these on its own. A "complete" protein has all
              9 in one place, like meat, eggs, dairy, soy, or quinoa. An "incomplete" one is
              missing some, like beans by themselves. That's not a problem — just eat a couple
              of incomplete ones together over the day (like beans and rice) and your body
              puts the pieces together just fine.
            </p>
            <p><strong style={{ color: COLORS.ink }}>For this dish: </strong>{n.protein_note}</p>
          </div>
        )}
      </div>

      <div className="flex gap-2 mt-4 flex-wrap">
        <button onClick={onSave}
          className="flex items-center gap-2 px-4 py-2 rounded-full text-sm font-medium"
          style={{ background: COLORS.gold, color: "#fff" }}>
          <Save size={14} /> Save this recipe
        </button>
        <button onClick={onAddToShopping}
          className="flex items-center gap-2 px-4 py-2 rounded-full text-sm font-medium"
          style={{ background: "transparent", border: `1px solid ${COLORS.clay}`, color: COLORS.clay }}>
          <ShoppingBasket size={14} /> Add to shopping list
        </button>
      </div>
    </div>
  );
}

function Field({ label, children, full }) {
  return (
    <div className={full ? "col-span-2" : ""}>
      <label style={{ fontSize: "0.75rem", color: COLORS.inkSoft, display: "block", marginBottom: "0.25rem" }}>
        {label}
      </label>
      {children}
    </div>
  );
}

function Stat({ label, value, color }) {
  return (
    <div className="rounded-xl p-3 text-center" style={{ background: COLORS.surface, border: `1px solid ${COLORS.line}` }}>
      <div className="display" style={{ fontSize: "1.4rem", color, fontWeight: 600 }}>{value}</div>
      <div style={{ fontSize: "0.7rem", color: COLORS.inkSoft }}>{label}</div>
    </div>
  );
}

function Fold({ title, icon, color, defaultOpen, children }) {
  const [open, setOpen] = useState(!!defaultOpen);
  const IconC = icon;
  return (
    <section className="rounded-2xl overflow-hidden"
      style={{ border: `1px solid ${COLORS.line}`, background: COLORS.surface }}>
      <button onClick={() => setOpen(!open)}
        className="w-full flex items-center justify-between px-4 py-3 text-left">
        <span className="flex items-center gap-2">
          {IconC && <IconC size={18} color={color || COLORS.gold} />}
          <span className="display" style={{ fontSize: "1.1rem", fontWeight: 600 }}>{title}</span>
        </span>
        <ChevronDown size={18} color={COLORS.inkSoft}
          style={{ transform: open ? "rotate(180deg)" : "none", transition: "transform 0.2s", flexShrink: 0 }} />
      </button>
      {open && <div className="px-4 pb-4">{children}</div>}
    </section>
  );
}

function FilterPill({ active, onClick, label }) {
  return (
    <button onClick={onClick} className="px-3 py-1 rounded-full text-xs font-medium"
      style={{
        background: active ? COLORS.ink : "transparent",
        color: active ? "#fff" : COLORS.inkSoft,
        border: `1px solid ${active ? COLORS.ink : COLORS.line}`,
      }}>
      {label}
    </button>
  );
}
